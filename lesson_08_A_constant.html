def Km(D1,D2,nc):
    K1=(nc-1)/D1
    K2=(nc-1)/D2
    return (K1+K2)/2

def z(Km,F):
    return Km+F/(1-F*0.012)
    
def real_d(L,n,z,D):
    d_part_1=1/2*(L+n/z)
    d_part_2=1/4*(L+n/z)**2
    d_part_3=L*n/z-n**2/(D*z)+n*L/D
    d = d_part_1-np.sqrt(d_part_2-d_part_3)
    return d
